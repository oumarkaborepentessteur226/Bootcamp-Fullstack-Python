class Farm:
    def __init__(self,nameF):
        self.list={}
        self.nameF=nameF


    def add_animal(self,name,nb):
            self.list.update({name:nb})


    def get_info(self):
        print(f"{self.nameF} farm")
        print(self.list)
        
        
    def    get_animal_types(self):
        print(f" la liste des animaux {list(self.list)}")
        
        
    def get_short_info(self):
        print(f"La ferme {self.nameF} poss√®de des {list(self.list)[0]}, des {list(self.list)[1]} et des {list(self.list)[2]} ")
